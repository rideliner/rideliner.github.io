def allow_addr?(addr)
  case @order
  when :deny_allow
    deny_allow?(addr)
  when :allow_deny
    allow_deny?(addr)
  else
    false
  end
end

def install_list(list)
  Hash[*list].each do |permission, domain|
    domain = Entry.new(domain)
    case permission.downcase
    when 'allow'
      @allow.push(domain)
    when 'deny'
      @deny.push(domain)
    else
      raise ArgumentError, "Invalid ACL entry #{list}"
    end
  end
end
